##  统计数组中每个数出现的次数

```javascript
      var arr = [
        1,
        1,
        1,
        2,
        3,
        4,
        4,
        5,
        6,
        6,
        5,
        5,
        6,
        7,
        88,
        55,
        111,
        22,
        11,
        44,
        44444,
      ];
      var obj = {};
      // 遍历数组，每遍历一个数字就看看这个数字是不是已经称为obj的属性
      for (var i = 0; i < arr.length - 1; i++) {
        // 如果是obj的属性，就让它加1
        if (obj.hasOwnProperty(arr[i])) {
          obj[arr[i]]++;
        } else {
          // 如果不是obj的属性，那么设置这个数字个数是1；
          obj[arr[i]] = 1;
        }
      }
      //   {1: 3, 2: 1, 3: 1, 4: 2, 5: 3, 6: 3, 7: 1, 11: 1, 22: 1, 44: 1, 55: 1, 88: 1, 111: 1}
      console.log(obj);
      for (var k in obj) {
        console.log(`数字${k}出现了${obj[k]}次`);
      }
```



```javascript
      function fun() {
        var a = 1;
        return {
          add: function () {
            a++;
          },
          f: function () {
            return this.a + a;
          },
        };
      }
      //   给函数本身添加一个属性a值是20
      fun.a = 20;
      var o = fun(); //返回一个对象
      //   给对象本身添加一个属性a值是30
      o.a = 30;
      o.add(); //a由1变为2
      var result = o.f(); //this.a指的是o对象的a,a值得是函数里面定义的a不是属性a
      console.log(result); //32
```

##  关于上下文的综合题目

```javascript
      var number = 2;
      var obj = {
        number: 4,
        fn1: (function () {
          var number;
          this.number *= 2; //这里this指window
          number = number * 2; //undefined
          var number = 3;
          return function () {
            this.number *= 2;
            number *= 3;
            alert(number);
          };
        })(),
        fn2: function () {
          this.number *= 2;
        },
      };
      var fn1 = obj.fn1;
      alert(number); //4
      fn1(); //this指全局，弹出9，fn1被调用一次，函数里的number已经变为9了
      obj.fn1(); //27
      obj.fn2(); //16
      alert(window.number); //8
      alert(obj.number); //16
```
