## 三、动态键

* ```javascript
  var obj = {
      a: 1,
      b: 2,
      c: 3
  }
  var k = 'c';
  console.log(obj.k); // undefined
  console.log(obj[k]); // 3
  console.log(obj['k']); // undefined
  ```

* k是变量，不能打点调用k

* **变量当作属性，不能被打点调用，必须使用方括号才能访问其值，变量不能加引号，否则就变成了字符串**

* ```javascript
  var obj = {
      a: 1,
      b: 2,
      c: 3,
      k: 4
  }
  var k = 'c';
  console.log(obj.k); // 4
  console.log(obj[k]); // 3
  console.log(obj['k']); // 4
  ```

## 四、对象的遍历（重要）

* ```javascript
  var obj = {
      a: 1,
      b: 2,
      c: 3,
      d: 4,
      e: 5,
      f: 6,
  };
  for (var k in obj) {
      // console.log(`${k}的值是${obj[k]}`);
      console.log(k + '的值是' + obj[k]);
      console.log(k);  //   a,b,c,d,e,f;表示键
      console.log(obj[k]);  // 1,2,3,4,5,6；表示值
  }
  ```

## 五、工作常用对象计算

* 对象不能排序，但是数组可以

```javascript
var obj={
    "中国":123,
    "美国":456,
    "法国":12,
    "德国":3,
    "英国":1234,
    "韩国":10
};
// 将对象转换为数组
var arr=[];
for(var k in obj){
    // 数组里面插入一个对象的写法
    arr.push({"country":k,"number":obj[k]});
}
// 将数组进行排序 
arr.sort(function(a,b){
    return a.number-b.number;//从小到大排序， b.number-a.number表示大到小排序
})
console.log(arr);
```

## 六、浅克隆

* 如果要克隆对象，必须要for循环

* 浅克隆只克隆一层，例如一个对象里面数组，那么它只分开对象这一层，实际上对象里面的数组还是指向同一个地址。

* ```javascript
  var obj1 = {
      a: 1,
      b: 2,
      c: [3, 5],
      d: 4
  }
  var obj2 = {};
  for (var k in obj1) {
      obj2[k] = obj1[k]
  }
  console.log(obj1 == obj2);    //false
  console.log(obj1.c == obj2.c);    //true
  ```

* 

### 1、数组浅克隆

```javascript
var arr1 = [1, 2, 3, 4, 5];
var arr2 = [];
for (var i = 0; i < arr1.length; i++) {
    arr2.push(arr1[i]);
}
console.log(arr2);// [1, 2, 3, 4, 5]
console.log(arr1 == arr2) //false
```

###  2、对象浅克隆

```javascript
var obj1 = {
    a: 1,
    b: 2,
    c: 3,
    d: 4
}
var obj2 = {};
for (var k in obj1) {
    //将obj1的键给obj2,obj1的值给obj2
    obj2[k] = obj1[k]
}
console.log(obj2);  //{ a: 1, b: 2, c: 3, d: 4 }
```

## 七、深克隆--递归（面试必考）

* **递归：**函数内部自己调用自己就是递归

* ```javascript
  function  factorial (n){
      if(n==1){
          return n;
      }
      return n*factorial (n-1);
  }
  var value=factorial (5);
  console.log(value);
  ```

* 递归一定要有结束条件。

* ```javascript
  function deepClone(m) {
      if (Array.isArray(m)) {
          var result = [];
          // 判断是不是数组
          for (var i = 0; i < m.length; i++) {
              result.push(deepClone(m[i]));
          }
      }
      else if (typeof m == 'object') {
          // 判断是不是对象
          var result = {};
          for (var k in m) {
              result[k] = deepClone(m[k]);
          }
      }
      else {
          var result = m;
      }
      return result;
  }
  var arr = [1, 2, 3, 4, 5, [12, 34, 123, { 'a': 4567, 'b': 4567 }]];
  var arr1 = deepClone(arr);
  var arr2 = arr1[5];
  var arr3 = arr2[3];
  var _arr2 = arr[5];
  var _arr3 = _arr2[3];
  
  console.log(arr1);
  console.log(arr == arr1); // false
  console.log(arr[5] === arr1[5]);  // false
  console.log(arr3 == _arr3);  // false
  ```
